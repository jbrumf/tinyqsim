 23class QCircuit(object):
 24    """ Class representing a quantum circuit.
 25
 26        QCircuit provides the main user interface for TinyQsim.
 27        A QCircuit instance represents a quantum circuit, with methods to add
 28        gates, query the state, plot data and perform quantum measurements.
 29
 30        The big-endian qubit convention is used.
 31    """

 33    def __init__(self, nqubits: int, init='zeros', auto_exec=True) -> None:
 34        """Initialize QCircuit.
 35           :param: nqubits: number of qubits
 36           :param: init: initialization mode: 'zeros' or 'random'
 37           :param: auto_exec: Enable on-the-fly execution
 38        """
 39        self._nqubits = nqubits
 40        self._init = init
 41        self._auto_exec = auto_exec
 42        self._model = Model(nqubits)
 43        self._schematic = Schematic(nqubits)
 44        self._simulator = Simulator(nqubits, init)
 45        self._gates = gates.GATES

 49    @property
 50    def state_vector(self) -> np.ndarray:
 51        """Return a copy of the quantum state vector.
 52           :return: copy of the quantum state vector
 53        """
 54        return self._simulator.state_vector

 56    @state_vector.setter
 57    def state_vector(self, state: ndarray) -> None:
 58        """Setter for state_vector.
 59            :param state: state vector
 60        """
 61        if not isclose(norm(state), 1.0):
 62            raise ValueError(f'State vector must be normalised: {norm(state)}')
 63        if not len(state) == 2 ** self._nqubits:
 64            raise ValueError(f'State vector should have length {2 ** self._nqubits}')
 65
 66        self._simulator.state_vector = state

 70        """Return the number of qubits.
 71           :return: number of qbits
 72        """
 73        return self._nqubits

 76        """Return the most recent results of measurements."""
 77        return self._simulator.results()

 81    def _check_qubits(self, qubits) -> None:
 82        """ Validate qubit indices.
 83            :param qubits: list of qubits
 84        """
 85        if min(qubits) < 0 or max(qubits) >= self._nqubits:
 86            raise ValueError(f'Qubit indices out of range: {qubits}')
 87
 88    def _add_gate(self, name: str, qubits: list[int], params: dict = None) -> None:
 89        """ Add a gate to the model.
 90            :param name: name of the gate
 91            :param qubits: qubits
 92            :param params: parameter dictionary
 93        """
 94        self._check_qubits(qubits)
 95        self._model.add_gate(name, qubits, params)
 96        if self._auto_exec:
 97            self._simulator.apply(self._gates[name], qubits)
 98
 99    def _add_param_gate(self, name: str, qubits: list[int], params: dict) -> None:
100        """ Add a parameterized gate to the model.
101            :param name: name of the gate
102            :param qubits: Qubits
103            :param params: Parameter dictionary
104        """
105        self._check_qubits(qubits)
106        self._model.add_gate(name, qubits, params)
107        if self._auto_exec:
108            u = self._gates[name](params['args'])
109            self._simulator.apply(u, qubits)
110
111    def _add_measure(self, qubits: list[int]) -> ndarray:
112        """ Add a measurement to the model.
113            :param qubits: list of qubits
114        """
115        self._model.add_gate('measure', qubits)
116        if self._auto_exec:
117            return self._simulator.measure(qubits)
118
119    def _add_reset(self, qubit: int) -> None:
120        """ Add a reset to the model.
121            :param qubit: qubit to be reset
122        """
123        self._model.add_gate('reset', [qubit])
124        if self._auto_exec:
125            self._simulator.reset(qubit)
126
127    def _add_unitary(self, name: str, u: ndarray, qubits: list[int], params: dict) -> None:
128        """ Add a unitary matrix to the model.
129            :param name: name of the gate
130            :param u: unitary matrix
131            :param qubits: qubits
132            :param params: Parameter dictionary
133        """
134        self._check_qubits(qubits)
135        if 2 ** len(qubits) != len(u):
136            raise ValueError(f'Wrong number of qubit indices, expected {quantum.n_qubits(u)}')
137
138        params['label'] = name
139        params['unitary'] = u
140        self._model.add_gate('U', qubits, params)
141        if self._auto_exec:
142            if not utils.is_unitary(u):
143                raise ValueError('Matrix must be unitary')
144            self._simulator.apply(u, qubits)

154    def components(self, decimals: int = 5, include_zeros: bool = False) -> dict:
155        """Return complex components of state vector as a dictionary.
156           :param decimals: number of decimal places (default=5)
157           :param include_zeros: True to include zero values (default=False)
158           :return: Dictionary of state components
159        """
160        comp = quantum.components_dict(self._simulator.state_vector)
161        return {k: round_complex(v, decimals) for k, v in comp.items() if include_zeros or abs(v) > EPS}
162
163    def probabilities(self, *qubits: int, decimals: int = 5,
164                      include_zeros: bool = False) -> dict[str, float]:
165        """ Return dictionary of the probabilities of each outcome.
166            :param qubits: qubits (None => all)
167            :param decimals: number of decimal places (default=5)
168            :param include_zeros: True to include zero values (default=False)
169            :return: dictionary of outcome->probability
170        """
171        if not qubits:
172            qubits = range(self._nqubits)
173        probs = quantum.probability_dict(self._simulator.state_vector, qubits)
174        return {k: round(v, decimals) for k, v in probs.items() if include_zeros or v > EPS}
175
176    def counts(self, *qubits: int, runs: int = 1000, rerun: bool = False,
177               include_zeros: bool = False) -> dict[str, int]:
178        """ Return measurement counts for repeated experiment.
179            Any measurements within the circuit will be re-run if rerun=True.
180            :param qubits: qubits (None => all)
181            :param runs: Number of test runs (default=1000)
182            :param rerun: True to re-run the whole experiment (default=False)
183            :param include_zeros: True to include zero values (default=False)
184            :return: frequencies of outcomes as a dictionary
185        """
186        if not qubits:
187            qubits = range(self.n_qubits)
188        if not rerun:
189            dic = quantum.counts_dict(self._simulator.state_vector, list(qubits), runs)
190        else:
191            sim = self._simulator
192            names = quantum.basis_names(len(qubits))
193            count = np.zeros(2 ** len(qubits), dtype=int)
194            for run in range(runs):
195                sim.execute(self._model)
196                probs = quantum.probabilities(sim.state_vector, qubits)
197                k = np.random.choice(len(probs), None, p=probs)
198                count[k] += 1
199            dic = dict(zip(names, count))
200        return {k: v for k, v in dic.items() if include_zeros or v > EPS}

188            qubits = range(self.n_qubits)

191        else:
192            sim = self._simulator
193            names = quantum.basis_names(len(qubits))
194            count = np.zeros(2 ** len(qubits), dtype=int)
195            for run in range(runs):
196                sim.execute(self._model)
197                probs = quantum.probabilities(sim.state_vector, qubits)
198                k = np.random.choice(len(probs), None, p=probs)
199                count[k] += 1
200            dic = dict(zip(names, count))
201        return {k: v for k, v in dic.items() if include_zeros or v > EPS}

Class representing a quantum circuit.

QCircuit provides the main user interface for TinyQsim. A QCircuit instance represents a quantum circuit, with methods to add gates, query the state, plot data and perform quantum measurements.

The big-endian qubit convention is used.
